def draw(dc)
  apply_dc(dc) do
    oldLineWidth = dc.lineWidth
    dc.lineWidth = 5 if selected?
    dc.drawArc(x, y, width, height,      0, 64*180)
    dc.drawArc(x, y, width, height, 64*180, 64*360)
    dc.lineWidth = oldLineWidth
  end
end
