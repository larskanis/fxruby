def drawBackground(table, dc, x, y, w, h)
  hg = table.horizontalGridShown? ? 1 : 0
  vg = table.verticalGridShown? ? 1 : 0
  dc.fillRectangle(x + vg, y + hg, w - vg, h - hg)
end
